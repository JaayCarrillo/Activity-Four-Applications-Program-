using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace Seconds_Elapsed
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void buttonConvert_Click(object sender, EventArgs e)
        {
           
            
            if(textBoxSeconds.Text == "")
            {
                MessageBox.Show("Please enter the value.");
            }
            else
            {
               
                int seconds = Convert.ToInt32(textBoxSeconds.Text);
                if(seconds >= 86400)
                {
                    int days = seconds / 86400;
                    textBoxResultLabel.Text = days + " days";
                }
               
                else if (seconds >= 3600)
                {
                    int hours = seconds / 3600;
                    textBoxResultLabel.Text = hours + " hours ";
                }

               
                else if(seconds >= 60)
                {
                    int minutes = seconds / 60;
                    textBoxResultLabel.Text = minutes + "minutes";
                }

                
                else
                {
                    textBoxResultLabel.Text = seconds + " seconds";
                }
            }
        }

        private void label1_Click(object sender, EventArgs e)
        {

        }

        private void label2_Click(object sender, EventArgs e)
        {

        }

        private void button1_Click(object sender, EventArgs e)
        {
            textBoxSeconds.Text = "";
            textBoxResultLabel.Text = "";

            // set focus 

            textBoxSeconds.Focus();
        }

        private void exitButton_Click(object sender, EventArgs e)
        { // close program 
            this.Close();
        }
    }
